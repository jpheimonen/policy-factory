from dataclasses import dataclass, field

from policy_factory.agent.tools import FILE_TOOLS, READ_ONLY_TOOLS

DEFAULT_MODELS = {

    "seed": "claude-sonnet-4-20250514",
    "values_seed": "claude-sonnet-4-20250514",
    "idea_evaluate": "claude-sonnet-4-20250514",
    "idea_generate": "claude-sonnet-4-20250514",
}

# Environment variable overrides
MODEL_ENV_VARS = {
    "generator": "POLICY_FACTORY_GENERATOR_MODEL",
    "critic": "POLICY_FACTORY_CRITIC_MODEL",
    "synthesis": "POLICY_FACTORY_SYNTHESIS_MODEL",
    "classifier": "POLICY_FACTORY_CLASSIFIER_MODEL",
    "heartbeat_skim": "POLICY_FACTORY_HEARTBEAT_SKIM_MODEL",
    "heartbeat_triage": "POLICY_FACTORY_HEARTBEAT_TRIAGE_MODEL",
    "heartbeat_update": "POLICY_FACTORY_HEARTBEAT_UPDATE_MODEL",
    "seed": "POLICY_FACTORY_SEED_MODEL",
    "values_seed": "POLICY_FACTORY_VALUES_SEED_MODEL",
    "idea_evaluate": "POLICY_FACTORY_IDEA_EVALUATE_MODEL",
    "idea_generate": "POLICY_FACTORY_IDEA_GENERATE_MODEL",
}


def resolve_model(role: str) -> str:
    """
    Resolve the model for an agent role.

    Checks environment variable first, falls back to default.
    """
    if role not in DEFAULT_MODELS:
        raise ValueError(f"Unknown agent role: {role}")

    env_var = MODEL_ENV_VARS.get(role)
    if env_var:
        override = os.environ.get(env_var)
        if override:
            return override

    return DEFAULT_MODELS[role]


# Tool sets by agent type
TOOL_SETS = {
    "generator": FILE_TOOLS,
    "critic": READ_ONLY_TOOLS,
    "synthesis": [],
    "classifier": [],
    "heartbeat_skim": [{"type": "web_search_20250305"}],
    "heartbeat_triage": [{"type": "web_search_20250305"}],
    "heartbeat_update": FILE_TOOLS + [{"type": "web_search_20250305"}],
    "seed": FILE_TOOLS + [{"type": "web_search_20250305"}],
    "values_seed": [],  # Uses Claude's knowledge, no tools
    "idea_evaluate": [],
    "idea_generate": [],
}


def get_tools_for_role(role: str) -> list[dict[str, Any]]:
    """Get the tool definitions for an agent role."""
    return TOOL_SETS.get(role, [])


@dataclass
class AgentConfig:
    """Configuration for an agent session."""

    role: str
    system_prompt: str = ""
    max_tokens: int = 4096

    # Resolved at creation time
    model: str = field(default="")
    tools: list[dict[str, Any]] = field(default_factory=list)

    def __post_init__(self):
        if not self.model:
            self.model = resolve_model(self.role)
        if not self.tools:
            self.tools = get_tools_for_role(self.role)


def create_generator_config(layer: str, system_prompt: str = "") -> AgentConfig:
    """Create config for a layer generator agent."""
    return AgentConfig(role="generator", system_prompt=system_prompt)


def create_critic_config(archetype: str, system_prompt: str = "") -> AgentConfig:

def create_classifier_config(system_prompt: str = "") -> AgentConfig:

def create_heartbeat_skim_config(system_prompt: str = "") -> AgentConfig:

def create_heartbeat_triage_config(system_prompt: str = "") -> AgentConfig:

def create_heartbeat_update_config(system_prompt: str = "") -> AgentConfig:

def create_seed_config(system_prompt: str = "") -> AgentConfig:

def create_values_seed_config(system_prompt: str = "") -> AgentConfig:
