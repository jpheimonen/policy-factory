# Step 003: Rewrite AgentSession Core

## Objective

Replace the `_run_once()` method in `AgentSession` with a proper Anthropic SDK implementation using `messages.stream()` for real-time token streaming. This is the core of the agent framework rewrite.

## Changes Required

### src/policy_factory/agent/session.py

Replace the entire `_run_once()` implementation and supporting code:

```python
"""
AgentSession - wraps Anthropic SDK with streaming, tool use, and retries.
"""

import asyncio
import json
import logging
from typing import Any

from anthropic import AsyncAnthropic, APIError, RateLimitError, APIStatusError

from policy_factory.agent.config import AgentConfig
from policy_factory.agent.errors import AgentError, ContextOverflowError
from policy_factory.agent.meditation_filter import MeditationFilter
from policy_factory.agent.tools import list_files, read_file, write_file, delete_file
from policy_factory.events import AgentTextChunk, EventEmitter

logger = logging.getLogger(__name__)

MAX_RETRIES = 3
BASE_DELAY = 1.0  # seconds


class AgentResult:
    """Result from an agent run."""

    def __init__(
        self,
        full_output: str,
        is_error: bool = False,
        error_message: str | None = None,
    ):
        self.full_output = full_output
        self.is_error = is_error
        self.error_message = error_message


class AgentSession:
    """
    Manages an agent conversation with streaming and tool use.
    """

    def __init__(
        self,
        config: AgentConfig,
        event_emitter: EventEmitter,
        context_id: str,
        agent_label: str,
        client: AsyncAnthropic,
        data_dir: str,
    ):
        self.config = config
        self.event_emitter = event_emitter
        self.context_id = context_id
        self.agent_label = agent_label
        self.client = client
        self.data_dir = Path(data_dir)
        self.meditation_filter = MeditationFilter()

    async def run(self, prompt: str) -> AgentResult:
        """
        Run the agent with the given prompt.

        Streams responses, handles tool calls, and retries on transient errors.
        """
        for attempt in range(MAX_RETRIES):
            try:
                return await self._run_conversation(prompt)
            except (RateLimitError, APIStatusError) as e:
                if self._is_transient_error(e) and attempt < MAX_RETRIES - 1:
                    delay = BASE_DELAY * (2 ** attempt)
                    logger.warning(
                        f"Transient error (attempt {attempt + 1}/{MAX_RETRIES}), "
                        f"retrying in {delay}s: {e}"
                    )
                    await asyncio.sleep(delay)
                    continue
                raise AgentError(
                    agent_role=self.config.role,
                    cascade_id=self.context_id,
                    message=str(e),
                ) from e
            except APIError as e:
                if "context_length" in str(e).lower() or "too many tokens" in str(e).lower():
                    raise ContextOverflowError(str(e)) from e
                raise AgentError(
                    agent_role=self.config.role,
                    cascade_id=self.context_id,
                    message=str(e),
                ) from e

        raise AgentError(
            agent_role=self.config.role,
            cascade_id=self.context_id,
            message="Max retries exceeded",
        )

    async def _run_conversation(self, prompt: str) -> AgentResult:
        """
        Run the agentic conversation loop until completion.
        """
        messages = [{"role": "user", "content": prompt}]
        full_output = []

        while True:
            # Stream the next turn
            turn_text, tool_calls, stop_reason = await self._stream_turn(messages)
            full_output.append(turn_text)

            # Add assistant response to conversation
            assistant_content = []
            if turn_text:
                assistant_content.append({"type": "text", "text": turn_text})
            assistant_content.extend(tool_calls)

            if assistant_content:
                messages.append({"role": "assistant", "content": assistant_content})

            # If no tool calls or end_turn, we're done
            if not tool_calls or stop_reason == "end_turn":
                break

            # Execute tool calls and add results
            tool_results = await self._execute_tools(tool_calls)
            messages.append({"role": "user", "content": tool_results})

        return AgentResult(full_output="\n".join(full_output))

    async def _stream_turn(
        self, messages: list[dict]
    ) -> tuple[str, list[dict], str | None]:
        """
        Stream a single turn, emitting text chunks and collecting tool calls.

        Returns (accumulated_text, tool_call_blocks, stop_reason).
        """
        text_chunks = []
        tool_calls = []
        current_tool: dict | None = None
        stop_reason = None

        async with self.client.messages.stream(
            model=self.config.model,
            max_tokens=self.config.max_tokens,
            system=self.config.system_prompt,
            messages=messages,
            tools=self.config.tools,
        ) as stream:
            async for event in stream:
                if event.type == "content_block_start":
                    if event.content_block.type == "tool_use":
                        current_tool = {
                            "type": "tool_use",
                            "id": event.content_block.id,
                            "name": event.content_block.name,
                            "input": "",
                        }

                elif event.type == "content_block_delta":
                    if event.delta.type == "text_delta":
                        text = event.delta.text
                        text_chunks.append(text)

                        # Apply meditation filter and emit
                        filtered = self.meditation_filter.process(text)
                        if filtered:
                            await self.event_emitter.emit(
                                AgentTextChunk(
                                    context_id=self.context_id,
                                    agent_label=self.agent_label,
                                    text=filtered,
                                )
                            )

                    elif event.delta.type == "input_json_delta":
                        if current_tool:
                            current_tool["input"] += event.delta.partial_json

                elif event.type == "content_block_stop":
                    if current_tool:
                        # Parse accumulated JSON input
                        try:
                            current_tool["input"] = json.loads(current_tool["input"])
                        except json.JSONDecodeError:
                            current_tool["input"] = {}
                        tool_calls.append(current_tool)
                        current_tool = None

                elif event.type == "message_delta":
                    stop_reason = event.delta.stop_reason

        return "".join(text_chunks), tool_calls, stop_reason

    async def _execute_tools(self, tool_calls: list[dict]) -> list[dict]:
        """
        Execute tool calls and return results for the conversation.
        """
        results = []

        for tool in tool_calls:
            name = tool["name"]
            args = tool["input"]
            tool_id = tool["id"]

            # Execute the tool
            if name == "list_files":
                result = list_files(args.get("directory", ""), self.data_dir)
            elif name == "read_file":
                result = read_file(args.get("path", ""), self.data_dir)
            elif name == "write_file":
                result = write_file(
                    args.get("path", ""),
                    args.get("content", ""),
                    self.data_dir,
                )
            elif name == "delete_file":
                result = delete_file(args.get("path", ""), self.data_dir)
            else:
                result = {"error": f"Unknown tool: {name}"}

            results.append({
                "type": "tool_result",
                "tool_use_id": tool_id,
                "content": json.dumps(result),
            })

        return results

    def _is_transient_error(self, error: Exception) -> bool:
        """Check if an error is transient and should be retried."""
        if isinstance(error, RateLimitError):
            return True
        if isinstance(error, APIStatusError):
            return error.status_code in (500, 502, 503, 529)
        return False
```

## Key Design Decisions

1. **Custom loop over tool_runner**: We use `messages.stream()` directly instead of the `tool_runner` beta because:
   - tool_runner is sync-only (our app is async)
   - tool_runner doesn't support streaming deltas
   - We need full control over tool execution for sandbox validation

2. **Streaming granularity**: Text deltas are emitted immediately as they arrive, passing through the meditation filter. This preserves the typewriter effect in the cascade viewer.

3. **Tool execution is sync**: File operations are fast enough that we don't need async file I/O. They execute between stream turns.

4. **Server-side tools (web_search)**: Handled automatically by the Anthropic API — they don't appear as tool_use blocks we need to execute.
