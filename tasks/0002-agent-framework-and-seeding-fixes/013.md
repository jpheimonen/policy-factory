"""
Tests for agent file tools.
"""

import pytest
from pathlib import Path

from policy_factory.agent.tools import (
    _validate_path,
    list_files,
    read_file,
    write_file,
    delete_file,
    FILE_TOOLS,
    READ_ONLY_TOOLS,
)


class TestPathValidation:
    """Test sandbox path validation."""

    def test_valid_path_within_sandbox(self, tmp_path):
        """Paths within data dir are allowed."""
        result = _validate_path("values/test.md", tmp_path)
        assert result == tmp_path / "values" / "test.md"

    def test_path_traversal_blocked(self, tmp_path):
        """Path traversal attempts are rejected."""
        with pytest.raises(ValueError, match="outside the data directory"):
            _validate_path("../etc/passwd", tmp_path)

    def test_double_traversal_blocked(self, tmp_path):
        """Multiple traversals are rejected."""
        with pytest.raises(ValueError, match="outside the data directory"):
            _validate_path("values/../../etc/passwd", tmp_path)

    def test_absolute_path_outside_blocked(self, tmp_path):
        """Absolute paths outside sandbox are rejected."""
        with pytest.raises(ValueError, match="outside the data directory"):
            _validate_path("/etc/passwd", tmp_path)

    def test_symlink_escape_blocked(self, tmp_path):
        """Symlinks that escape sandbox are blocked."""
        # Create a symlink pointing outside
        (tmp_path / "values").mkdir()
        link = tmp_path / "values" / "escape"
        link.symlink_to("/etc")

        with pytest.raises(ValueError, match="outside the data directory"):
            _validate_path("values/escape/passwd", tmp_path)


class TestListFiles:
    """Test list_files tool."""

    def test_list_markdown_files(self, tmp_path):
        """Returns markdown filenames."""
        values_dir = tmp_path / "values"
        values_dir.mkdir()
        (values_dir / "sovereignty.md").write_text("content")
        (values_dir / "security.md").write_text("content")
        (values_dir / "README.md").write_text("readme")

        result = list_files("values", tmp_path)

        assert "files" in result
        assert sorted(result["files"]) == ["security.md", "sovereignty.md"]
        assert "README.md" not in result["files"]

    def test_empty_directory(self, tmp_path):
        """Returns empty list for empty directory."""
        (tmp_path / "values").mkdir()

        result = list_files("values", tmp_path)

        assert result == {"files": []}

    def test_outside_sandbox_error(self, tmp_path):
        """Returns error for paths outside sandbox."""
        result = list_files("../etc", tmp_path)

        assert "error" in result
        assert "outside" in result["error"].lower()

    def test_not_a_directory(self, tmp_path):
        """Returns error for file path."""
        (tmp_path / "test.md").write_text("content")

        result = list_files("test.md", tmp_path)

        assert "error" in result


class TestReadFile:
    """Test read_file tool."""

    def test_read_existing_file(self, tmp_path):
        """Returns file content."""
        (tmp_path / "values").mkdir()
        test_content = "---\ntitle: Test\n---\nBody text"
        (tmp_path / "values" / "test.md").write_text(test_content)

        result = read_file("values/test.md", tmp_path)

        assert result == {"content": test_content}

    def test_file_not_found(self, tmp_path):
        """Returns error for nonexistent file."""
        result = read_file("values/nonexistent.md", tmp_path)

        assert "error" in result
        assert "not found" in result["error"].lower()

    def test_outside_sandbox_error(self, tmp_path):
        """Returns error for paths outside sandbox."""
        result = read_file("../etc/passwd", tmp_path)

        assert "error" in result


class TestWriteFile:
    """Test write_file tool."""

    def test_create_new_file(self, tmp_path):
        """Creates new file with content."""
        (tmp_path / "values").mkdir()
        content = "---\ntitle: New\n---\nContent"

        result = write_file("values/new.md", content, tmp_path)

        assert result == {"success": True, "path": "values/new.md"}
        assert (tmp_path / "values" / "new.md").read_text() == content

    def test_overwrite_existing_file(self, tmp_path):
        """Overwrites existing file."""
        (tmp_path / "values").mkdir()
        (tmp_path / "values" / "existing.md").write_text("old content")

        result = write_file("values/existing.md", "new content", tmp_path)

        assert result["success"]
        assert (tmp_path / "values" / "existing.md").read_text() == "new content"

    def test_creates_parent_directories(self, tmp_path):
        """Creates parent directories if needed."""
        result = write_file("new-layer/sub/file.md", "content", tmp_path)

        assert result["success"]
        assert (tmp_path / "new-layer" / "sub" / "file.md").exists()

    def test_outside_sandbox_error(self, tmp_path):
        """Returns error for paths outside sandbox."""
        result = write_file("../escape.md", "content", tmp_path)

        assert "error" in result


class TestDeleteFile:
    """Test delete_file tool."""

    def test_delete_existing_file(self, tmp_path):
        """Deletes existing file."""
        (tmp_path / "values").mkdir()
        (tmp_path / "values" / "delete-me.md").write_text("content")

        result = delete_file("values/delete-me.md", tmp_path)

        assert result == {"success": True, "path": "values/delete-me.md"}
        assert not (tmp_path / "values" / "delete-me.md").exists()

    def test_nonexistent_file_succeeds(self, tmp_path):
        """Succeeds silently for nonexistent file (idempotent)."""
        result = delete_file("values/nonexistent.md", tmp_path)

        assert result["success"]

    def test_outside_sandbox_error(self, tmp_path):
        """Returns error for paths outside sandbox."""
        result = delete_file("../escape.md", tmp_path)

        assert "error" in result


class TestToolDefinitions:
    """Test tool definition structures."""

    def test_file_tools_complete(self):
        """FILE_TOOLS contains all four tools."""
        names = {t["name"] for t in FILE_TOOLS}
        assert names == {"list_files", "read_file", "write_file", "delete_file"}

    def test_read_only_tools_subset(self):
        """READ_ONLY_TOOLS is read-only subset."""
        names = {t["name"] for t in READ_ONLY_TOOLS}
        assert names == {"list_files", "read_file"}

    def test_tool_schemas_valid(self):
        """All tools have required schema fields."""
        for tool in FILE_TOOLS:
            assert "name" in tool
            assert "description" in tool
            assert "input_schema" in tool
            assert tool["input_schema"]["type"] == "object"
