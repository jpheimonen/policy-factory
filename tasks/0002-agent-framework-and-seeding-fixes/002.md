# Step 002: Create File Tools Module

## Objective

Create a new module providing file operation tools that agents can invoke. All tools enforce sandbox boundaries to prevent access outside the `data/` directory.

## Changes Required

### New file: src/policy_factory/agent/tools.py

Create tools that agents can call:

```python
"""
Agent file tools with sandbox validation.

All tools restrict access to the configured data directory.
"""

from pathlib import Path
from typing import Any

from policy_factory.data.markdown import read_markdown, write_markdown


def _validate_path(path: str, data_dir: Path) -> Path:
    """
    Validate that a path resolves within the data directory.

    Raises ValueError if path escapes sandbox.
    """
    resolved = (data_dir / path).resolve()
    if not resolved.is_relative_to(data_dir.resolve()):
        raise ValueError(f"Path {path} is outside the data directory")
    return resolved


def list_files(directory: str, data_dir: Path) -> dict[str, Any]:
    """List markdown files in a layer directory, excluding README.md."""
    try:
        resolved = _validate_path(directory, data_dir)
        if not resolved.is_dir():
            return {"error": f"Not a directory: {directory}"}

        files = [
            f.name for f in resolved.glob("*.md")
            if f.name != "README.md"
        ]
        return {"files": sorted(files)}
    except ValueError as e:
        return {"error": str(e)}


def read_file(path: str, data_dir: Path) -> dict[str, Any]:
    """Read a markdown file's contents including frontmatter."""
    try:
        resolved = _validate_path(path, data_dir)
        if not resolved.exists():
            return {"error": f"File not found: {path}"}

        content = resolved.read_text()
        return {"content": content}
    except ValueError as e:
        return {"error": str(e)}


def write_file(path: str, content: str, data_dir: Path) -> dict[str, Any]:
    """Write content to a markdown file, creating parent dirs if needed."""
    try:
        resolved = _validate_path(path, data_dir)
        resolved.parent.mkdir(parents=True, exist_ok=True)
        resolved.write_text(content)
        return {"success": True, "path": path}
    except ValueError as e:
        return {"error": str(e)}


def delete_file(path: str, data_dir: Path) -> dict[str, Any]:
    """Delete a file. Succeeds silently if file doesn't exist (idempotent)."""
    try:
        resolved = _validate_path(path, data_dir)
        if resolved.exists():
            resolved.unlink()
        return {"success": True, "path": path}
    except ValueError as e:
        return {"error": str(e)}


# Tool definitions for Anthropic API
FILE_TOOLS = [
    {
        "name": "list_files",
        "description": "List markdown files in a layer directory (e.g., 'values', 'situational-awareness'). Returns filenames excluding README.md.",
        "input_schema": {
            "type": "object",
            "properties": {
                "directory": {
                    "type": "string",
                    "description": "The layer directory to list (e.g., 'values', 'situational-awareness', 'strategic', 'tactical', 'policies')"
                }
            },
            "required": ["directory"]
        }
    },
    {
        "name": "read_file",
        "description": "Read the contents of a markdown file, including YAML frontmatter.",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {
                    "type": "string",
                    "description": "Path to the file relative to data directory (e.g., 'values/sovereignty.md')"
                }
            },
            "required": ["path"]
        }
    },
    {
        "name": "write_file",
        "description": "Write content to a markdown file. Creates parent directories if needed. Overwrites if file exists.",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {
                    "type": "string",
                    "description": "Path for the file relative to data directory (e.g., 'values/sovereignty.md')"
                },
                "content": {
                    "type": "string",
                    "description": "The full content to write, including YAML frontmatter"
                }
            },
            "required": ["path", "content"]
        }
    },
    {
        "name": "delete_file",
        "description": "Delete a markdown file. Succeeds silently if the file doesn't exist.",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {
                    "type": "string",
                    "description": "Path to the file to delete relative to data directory"
                }
            },
            "required": ["path"]
        }
    }
]

# Read-only subset for critic agents
READ_ONLY_TOOLS = [t for t in FILE_TOOLS if t["name"] in ("list_files", "read_file")]
```
