### src/policy_factory/server/routers/seed.py

Update the status endpoint:

```python
class SeedStatusResponse(BaseModel):
    """Response for seed status endpoint."""
    values_seeded: bool
    values_count: int
    sa_seeded: bool
    sa_count: int


@router.get("/status", response_model=SeedStatusResponse)
async def get_seed_status(
    data_dir=Depends(get_data_dir),
):
    """
    Get the current seeding status for values and SA layers.

    Returns whether each layer has been seeded and the item counts.
    """
    # Check values layer
    values_items = list_items("values", data_dir)
    values_count = len(values_items)
    values_seeded = values_count > 0

    # Check SA layer
    sa_items = list_items("situational-awareness", data_dir)
    sa_count = len(sa_items)
    sa_seeded = sa_count > 0

    return SeedStatusResponse(
        values_seeded=values_seeded,
        values_count=values_count,
        sa_seeded=sa_seeded,
        sa_count=sa_count,
    )
```
