# Step 007: Implement Values Seed Endpoint

## Objective

Create a new endpoint to trigger values layer seeding. The agent synthesizes axiomatic values using Claude's knowledge, and the endpoint writes them directly to `data/values/`.

## Changes Required

### src/policy_factory/server/routers/seed.py

Add a new endpoint for values seeding:

```python
from pathlib import Path
import re

from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel

from policy_factory.agent.config import create_values_seed_config
from policy_factory.agent.session import AgentSession, AgentResult
from policy_factory.data.git import commit_changes
from policy_factory.data.layers import write_item, delete_item, list_items, write_narrative
from policy_factory.prompts.loader import load_prompt
from policy_factory.server.dependencies import (
    get_anthropic_client,
    get_data_dir,
    get_event_emitter,
    require_auth,
)


class ValuesSeedResponse(BaseModel):
    success: bool
    values_created: int
    message: str


@router.post("/values", response_model=ValuesSeedResponse)
async def seed_values(
    _user=Depends(require_auth),
    client=Depends(get_anthropic_client),
    data_dir=Depends(get_data_dir),
    event_emitter=Depends(get_event_emitter),
):
    """
    Seed the values layer with AI-synthesized axiomatic values.

    This clears any existing values and writes new ones based on
    Claude's knowledge of Finnish policy discourse.
    """
    # Load the values seed prompt
    prompt = load_prompt("seed/values")

    # Create agent config (no tools, no meditation)
    config = create_values_seed_config()

    # Run the agent
    session = AgentSession(
        config=config,
        event_emitter=event_emitter,
        context_id="values-seed",
        agent_label="Values Synthesizer",
        client=client,
        data_dir=data_dir,
    )

    result = await session.run(prompt)

    if result.is_error:
        raise HTTPException(status_code=500, detail=result.error_message)

    # Parse values from output
    values = parse_values_output(result.full_output)

    if not values:
        raise HTTPException(
            status_code=500,
            detail="Agent produced no parseable values"
        )

    # Clear existing values
    existing = list_items("values", data_dir)
    for item in existing:
        delete_item("values", item["slug"], data_dir)

    # Write new values
    for value in values:
        write_item(
            layer="values",
            slug=value["slug"],
            title=value["title"],
            body=value["body"],
            metadata={"tensions": value.get("tensions", [])},
            data_dir=data_dir,
        )

    # Regenerate narrative summary
    narrative = generate_values_narrative(values)
    write_narrative("values", narrative, data_dir)

    # Commit changes
    commit_changes(data_dir, f"Seed values layer with {len(values)} values")

    return ValuesSeedResponse(
        success=True,
        values_created=len(values),
        message=f"Created {len(values)} values",
    )


def parse_values_output(output: str) -> list[dict]:
    """
    Parse the agent's output into individual value documents.

    Expects markdown with YAML frontmatter blocks.
    """
    values = []

    # Split on frontmatter delimiters
    pattern = r'---\n(.*?)\n---\n(.*?)(?=\n---\n|$)'
    matches = re.findall(pattern, output, re.DOTALL)

    for frontmatter, body in matches:
        value = {"body": body.strip()}

        # Parse simple YAML frontmatter
        for line in frontmatter.strip().split('\n'):
            if ':' in line:
                key, val = line.split(':', 1)
                key = key.strip()
                val = val.strip()

                if key == 'title':
                    value['title'] = val
                    value['slug'] = slugify(val)
                elif key == 'tensions':
                    # Handle list format
                    value['tensions'] = []

        # Collect tension items (lines starting with -)
        for line in frontmatter.strip().split('\n'):
            if line.strip().startswith('-'):
                tension = line.strip()[1:].strip()
                if 'tensions' in value:
                    value['tensions'].append(tension)

        if 'title' in value and 'body' in value:
            values.append(value)

    return values


def slugify(title: str) -> str:
    """Convert title to URL-safe slug."""
    slug = title.lower()
    slug = re.sub(r'[^a-z0-9]+', '-', slug)
    slug = slug.strip('-')
    return slug


def generate_values_narrative(values: list[dict]) -> str:
    """Generate a narrative summary of the values layer."""
    lines = ["# Values Layer\n"]
    lines.append("This layer contains the foundational axiomatic values that ")
    lines.append("underpin Finnish policy discourse.\n\n")
    lines.append("## Values\n\n")

    for value in values:
        lines.append(f"- **{value['title']}**")
        if value.get('tensions'):
            lines.append(f" (tensions: {', '.join(value['tensions'])})")
        lines.append("\n")

    return "".join(lines)
```
