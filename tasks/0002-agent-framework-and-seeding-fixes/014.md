class TestSASeedingFlow:
    """Test SA seeding endpoint behavior."""

    @pytest.mark.asyncio
    async def test_sa_seed_no_409(self, integration_client, data_dir):
        """SA seeding no longer returns 409 when layer has items."""
        # Pre-populate SA layer
        from policy_factory.data.layers import write_item
        write_item("situational-awareness", "existing", "Existing", "content", {}, data_dir)

        with patch("policy_factory.agent.session.AgentSession.run") as mock_run:
            mock_run.return_value = AgentResult(full_output="Done")

            response = await integration_client.post(
                "/api/seed/",
                headers={"Authorization": "Bearer test-token"},
            )

        # Should succeed, not 409
        assert response.status_code == 200

    @pytest.mark.asyncio
    async def test_sa_seed_accepts_context(self, integration_client):
        """SA seeding accepts optional context parameter."""
        with patch("policy_factory.agent.session.AgentSession.run") as mock_run:
            mock_run.return_value = AgentResult(full_output="Done")

            response = await integration_client.post(
                "/api/seed/",
                headers={"Authorization": "Bearer test-token"},
                json={"context": "Current geopolitical situation..."},
            )

        assert response.status_code == 200

        # Verify context was included in prompt
        call_args = mock_run.call_args
        prompt = call_args[0][0] if call_args[0] else call_args[1].get("prompt", "")
        assert "geopolitical" in prompt.lower()

    async def test_sa_seed_triggers_cascade(self, integration_client):
        """SA seeding triggers a cascade after completion."""

class TestDataInitialization:
    """Test data directory initialization."""

    def test_init_creates_directories(self, tmp_path):
        """Initialization creates layer directories."""
        from policy_factory.data.init import initialize_data_directory

        data_dir = tmp_path / "data"
        initialize_data_directory(str(data_dir))

        assert (data_dir / "values").is_dir()
        assert (data_dir / "situational-awareness").is_dir()
        assert (data_dir / "strategic").is_dir()
        assert (data_dir / "tactical").is_dir()
        assert (data_dir / "policies").is_dir()

    def test_init_no_preseeded_values(self, tmp_path):
        """Initialization does not write pre-seeded values."""
        from policy_factory.data.init import initialize_data_directory

        data_dir = tmp_path / "data"
        initialize_data_directory(str(data_dir))

        # Values directory should exist but have no item files
        values_dir = data_dir / "values"
        md_files = list(values_dir.glob("*.md"))
        # Only README.md should exist
        assert len(md_files) <= 1
        if md_files:
            assert md_files[0].name == "README.md"

    def test_init_idempotent(self, tmp_path):
        """Initialization is safe to call multiple times."""
        from policy_factory.data.init import initialize_data_directory

        data_dir = tmp_path / "data"
        initialize_data_directory(str(data_dir))
        initialize_data_directory(str(data_dir))  # Should not error

        assert (data_dir / "values").is_dir()
